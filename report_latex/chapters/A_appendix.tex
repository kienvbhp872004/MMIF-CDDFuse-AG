\section{Reproduce instructions}
\label{app:reproduce}

\begin{verbatim}
# 1. Clone repo
git clone https://github.com/kienvbhp872004/Image-Fusion
cd Image-Fusion

# 2. Setup Python 3.8 venv
py -3.8 -m venv .venv38
.\.venv38\Scripts\Activate.ps1

# 3. Install deps
pip install -r models/MMIF-CDDFuse/requirements.txt
pip install torch==2.1.2+cu118 torchvision==0.16.2+cu118 \
    --index-url https://download.pytorch.org/whl/cu118

# 4. Preprocess
cd models/MMIF-CDDFuse
python dataprocessing_MIF.py

# 5. Train
python train_MIF.py --variant Combined-Gated-Saliency --amp

# 6. Evaluate
python evaluate_cddfuse.py --variant Combined-Gated-Saliency \
    --modal CT --ckpt models/CDDFuse-Combined-Gated-Saliency_MIF_*.pth \
    --harvard_root ../../data/reference --save_perimage
\end{verbatim}
